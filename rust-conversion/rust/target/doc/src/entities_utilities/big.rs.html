1//! Big Number Operations
2//!
3//! Provides arbitrary precision integer operations.
4//! Based on big.c
5//!
6//! This module uses the `malachite` crate for high-performance
7//! arbitrary-precision arithmetic, which provides behavior compatible
8//! with the C implementation's two's complement semantics for operations.
9
10use malachite::Integer;
11use malachite::num::arithmetic::traits::Abs;
12use malachite::num::conversion::traits::ToStringBase;
13
14/// Big number representation using malachite's Integer
15///
16/// This wraps malachite's Integer type to provide the same API
17/// as the C big.c implementation. Malachite uses two's complement
18/// representation internally, which matches the C code's bitwise
19/// operation semantics.
20#[derive(Clone, Debug, PartialEq, Eq, Hash)]
21pub struct BigNumber {
22    value: Integer,
23}
24
25impl BigNumber {
26    /// Create a new big number from i64
27    pub fn from_i64(value: i64) -> Self {
28        Self {
29            value: Integer::from(value),
30        }
31    }
32
33    /// Create a new big number from u64
34    pub fn from_u64(value: u64) -> Self {
35        Self {
36            value: Integer::from(value),
37        }
38    }
39
40    /// Create a new big number from u32
41    pub fn from_u32(value: u32) -> Self {
42        Self {
43            value: Integer::from(value),
44        }
45    }
46
47    /// Create a new big number from i32
48    pub fn from_i32(value: i32) -> Self {
49        Self {
50            value: Integer::from(value),
51        }
52    }
53
54    /// Create a new big number from double (f64)
55    ///
56    /// Returns None if the conversion fails (NaN, infinity, or out of range)
57    pub fn from_f64(value: f64) -> Option<Self> {
58        if !value.is_finite() {
59            return None;
60        }
61        // Convert to integer by truncating
62        // Note: C code does similar conversion in double_to_big
63        // Malachite doesn't have direct from_f64, so we convert via i64 first
64        // For values outside i64 range, we'd need a more complex conversion
65        let truncated = value.trunc();
66        if truncated >= i64::MIN as f64 && truncated <= i64::MAX as f64 {
67            Some(Self {
68                value: Integer::from(truncated as i64),
69            })
70        } else {
71            None
72        }
73    }
74
75    /// Convert to f64
76    ///
77    /// Returns None if the value is too large to represent as f64
78    pub fn to_f64(&self) -> Option<f64> {
79        // Try converting via i64 first (limited range)
80        if let Ok(i64_val) = self.value.clone().try_into() {
81            Some(i64_val as f64)
82        } else {
83            None
84        }
85    }
86
87    /// Convert to u32
88    ///
89    /// Returns None if the value is negative or too large
90    pub fn to_u32(&self) -> Option<u32> {
91        if self.value >= 0 {
92            self.value.clone().try_into().ok()
93        } else {
94            None
95        }
96    }
97
98    /// Convert to i64
99    ///
100    /// Returns None if the value is out of range
101    pub fn to_i64(&self) -> Option<i64> {
102        self.value.clone().try_into().ok()
103    }
104
105    /// Get the sign of the number
106    ///
107    /// Returns true if positive or zero, false if negative
108    pub fn is_positive(&self) -> bool {
109        self.value >= 0
110    }
111
112    /// Check if the number is zero
113    pub fn is_zero(&self) -> bool {
114        self.value == 0
115    }

117    /// Add two big numbers: x + y
118    pub fn plus(&self, other: &Self) -> Self {
119        Self {
120            value: &self.value + &other.value,
121        }
122    }
123
124    /// Subtract two big numbers: x - y
125    pub fn minus(&self, other: &Self) -> Self {
126        Self {
127            value: &self.value - &other.value,
128        }
129    }
130
131    /// Multiply two big numbers: x * y
132    pub fn times(&self, other: &Self) -> Self {
133        Self {
134            value: &self.value * &other.value,
135        }
136    }
137
138    /// Divide two big numbers: x / y
139    ///
140    /// Returns None if dividing by zero
141    pub fn div(&self, other: &Self) -> Option<Self> {
142        if other.is_zero() {
143            return None;
144        }
145        Some(Self {
146            value: &self.value / &other.value,
147        })
148    }
149
150    /// Remainder of division: x % y
151    ///
152    /// Returns None if dividing by zero
153    pub fn rem(&self, other: &Self) -> Option<Self> {
154        if other.is_zero() {
155            return None;
156        }
157        Some(Self {
158            value: &self.value % &other.value,
159        })
160    }
161
162    /// Multiply and add: x * y + z
163    pub fn mul_add(&self, y: &Self, z: &Self) -> Self {
164        Self {
165            value: &self.value * &y.value + &z.value,
166        }
167    }
168
169    /// Add a small unsigned integer: x + y
170    pub fn plus_small(&self, y: u32) -> Self {
171        Self {
172            value: &self.value + Integer::from(y),
173        }
174    }
175
176    /// Bitwise AND: x & y
177    pub fn bitand(&self, other: &Self) -> Self {
178        Self {
179            value: &self.value & &other.value,
180        }
181    }
182
183    /// Bitwise OR: x | y
184    pub fn bitor(&self, other: &Self) -> Self {
185        Self {
186            value: &self.value | &other.value,
187        }
188    }
189
190    /// Bitwise XOR: x ^ y
191    pub fn bitxor(&self, other: &Self) -> Self {
192        Self {
193            value: &self.value ^ &other.value,
194        }
195    }
196
197    /// Bitwise NOT: !x
198    ///
199    /// In Erlang, bnot -X == (X - 1) and bnot +X == -(X + 1)
200    /// This matches two's complement semantics
201    pub fn bitnot(&self) -> Self {
202        Self {
203            value: !&self.value,
204        }
205    }
206
207    /// Left shift: x << y
208    ///
209    /// y can be negative for right shift
210    pub fn lshift(&self, shift: i32) -> Self {
211        if shift >= 0 {
212            Self {
213                value: &self.value << shift as u64,
214            }
215        } else {
216            Self {
217                value: &self.value >> (-shift) as u64,
218            }
219        }
220    }
221
222    /// Compare two big numbers (signed comparison)
223    ///
224    /// Returns:
225    /// - -1 if self < other
226    /// - 0 if self == other
227    /// - 1 if self > other
228    pub fn comp(&self, other: &Self) -> i32 {
229        match self.value.cmp(&other.value) {
230            std::cmp::Ordering::Less => -1,
231            std::cmp::Ordering::Equal => 0,
232            std::cmp::Ordering::Greater => 1,
233        }
234    }
235
236    /// Unsigned comparison of two big numbers
237    ///
238    /// Compares absolute values, ignoring sign
239    /// Returns:
240    /// - -1 if |self| < |other|
241    /// - 0 if |self| == |other|
242    /// - 1 if |self| > |other|
243    pub fn ucomp(&self, other: &Self) -> i32 {
244        let abs_self = self.value.clone().abs();
245        let abs_other = other.value.clone().abs();
246        match abs_self.cmp(&abs_other) {
247            std::cmp::Ordering::Less => -1,
248            std::cmp::Ordering::Equal => 0,
249            std::cmp::Ordering::Greater => 1,
250        }
251    }
252
253    /// Convert to string representation in given base
254    ///
255    /// Base must be between 2 and 36
256    pub fn to_string_base(&self, base: u32) -> String {
257        if base < 2 || base > 36 {
258            panic!("Base must be between 2 and 36");
259        }
260        self.value.to_string_base(base as u8)
261    }
262
263    /// Get the internal Integer value (for advanced use)
264    pub fn as_integer(&self) -> &Integer {
265        &self.value
266    }
267
268    /// Create from Integer (for advanced use)
269    pub fn from_integer(value: Integer) -> Self {
270        Self { value }
271    }

302    #[test]
303    fn test_big_number_creation() {
304        let big = BigNumber::from_i64(12345);
305        assert!(big.is_positive());
306        assert!(!big.is_zero());
307
308        let big_neg = BigNumber::from_i64(-12345);
309        assert!(!big_neg.is_positive());
310    }

312    #[test]
313    fn test_arithmetic_operations() {
314        let a = BigNumber::from_i64(100);
315        let b = BigNumber::from_i64(50);
316
317        assert_eq!(a.plus(&b).to_i64(), Some(150));
318        assert_eq!(a.minus(&b).to_i64(), Some(50));
319        assert_eq!(a.times(&b).to_i64(), Some(5000));
320        assert_eq!(a.div(&b).unwrap().to_i64(), Some(2));
321        assert_eq!(a.rem(&b).unwrap().to_i64(), Some(0));
322    }

325    fn test_bitwise_operations() {
326        let a = BigNumber::from_i64(0b1010); // 10
327        let b = BigNumber::from_i64(0b1100); // 12
328
329        assert_eq!(a.bitand(&b).to_i64(), Some(0b1000)); // 8
330        assert_eq!(a.bitor(&b).to_i64(), Some(0b1110)); // 14
331        assert_eq!(a.bitxor(&b).to_i64(), Some(0b0110)); // 6
332    }
